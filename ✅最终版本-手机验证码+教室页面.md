function sendCode(type) {
    const phone = document.getElementById(type === 'login' ? 'login-phone' : 'register-phone').value;
    
    if (!phone || phone.length !== 11) {
        showNotification('请输入正确的手机号', 'error');
        return;
    }
    
    // 模拟发送验证码
    showNotification('验证码已发送（演示：123456）', 'success');

            // 显示作业内容
